   <script>  
           function validation()  
           {  
               var id=document.f1.username.value;
               var ph=document.f1.num.value;  
               var ps=document.f1.pswd.value;  
               var cps=document.f1.cpswd.value;

               const incorrectUsername = document.getElementById('incorrectUsername')
               const incorrectmobileNo = document.getElementById('incorrectmobileNo')
               const incorrectPassword = document.getElementById('incorrectPassword')
               const incorrectConfirmPassword = document.getElementById('incorrectConfirmPassword')
               
                if(id.length=="") {  
                    incorrectUsername.innerHTML = 'Enter User Name'; 
                    return false;  
                }else if(id.length >= 20) {  
                    incorrectUsername.innerHTML = 'User Name must be less than 20 Charecters'; 
                    return false;  
                }else{
                    incorrectUsername.innerHTML = '';
                }

                if (ph.length == "") {  
                    incorrectmobileNo.innerHTML = 'Enter Mobile Number';
                    return false;  
                }else if(ph.length != 10) {  
                    incorrectmobileNo.innerHTML = 'Enter Valid Mobile Number'; 
                    return false;  
                }else{
                    incorrectmobileNo.innerHTML = '';
                }

                
                if(ps.length=="") {  
                    incorrectPassword.innerHTML = 'Enter Password'; 
                    return false;  
                }else if(ps.length <= 8){
                    incorrectPassword.innerText = 'Password must be more then 8 Charecters'
                    return false; 
                }else if(ps.length >= 20){
                    incorrectPassword.innerText = 'Password must be less then 20 Charecters'
                    return false; 
                }else{
                    incorrectPassword.innerHTML = '';
                }


                if (cps.length=="") {  
                    incorrectConfirmPassword.innerHTML = 'Enter Confirm Password';
                    return false;  
                }else if(ps!=cps) {  
                    incorrectConfirmPassword.innerHTML = "Password and confirm password in not equal";  
                    return false;  
                }else{
                    incorrectConfirmPassword.innerHTML = ""; 
                }
            }                            
       </script>
